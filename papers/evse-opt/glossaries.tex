% Main glossary entries -- definitions of relevant terminology
\newglossaryentry{l1}{
    name=Level 1,
    description={Level 1 Charging uses standard 120 volts AC power, typically provides 2-5 miles of range per hour. Learn more \href{https://www.ti.com/lit/wp/sway014/sway014.pdf}{here}}
}

\newglossaryentry{l2}{
    name=Level 2,
    description={Level 2 Charging uses 240 volts AC power, typically provides 10-60 miles of range per hour. Learn more \href{https://www.ti.com/lit/wp/sway014/sway014.pdf}{here}}
}

\newglossaryentry{l3}{
    name=Level 3,
    description={Level 3 Charging, also known as DC Fast Charging, uses 480 volts DC power, typically provides 60-100 miles of range per hour. Learn more \href{https://www.ti.com/lit/wp/sway014/sway014.pdf}{here}}
}

\newglossaryentry{osm}{
    name=OpenStreetMap,
    description={A package used to easily download, model, analyze, and visualize street networks and other geo-spatial features. Learn more \href{www.openstreetmap.org}{here}.}
}

\newglossaryentry{row-open-data}{
    name=Region of Waterloo's Open Data Portal,
    description={An online platform provided by the Region of Waterloo offering public access to various datasets, including geographic and infrastructure data. Check it out \href{https://rowopendata-rmw.opendata.arcgis.com/}{here}.}
}

\newglossaryentry{ocm}{
    name=OpenChargeMap API,
    description={OpenChargeMap is a non-commercial, non-profit, \glsxtrshort{ev} data service. Their API can be used to retrieve charging location information, user comments, checkins and photos and to submit new comments/checkins. Learn more \href{https://openchargemap.org/site/develop}{here}.}
}

\newglossaryentry{utm-17n}{
    name=UTM Zone 17N Projection,
    description={The \glsxtrfull{utm} is a coordinate system that divides the world into a series of zones for mapping and spatial measurements, specifically referencing Zone 17N in our context. Learn more about it \href{https://epsg.io/32617}{here} or from \href{https://natural-resources.canada.ca/earth-sciences/geography/topographic-information/maps/utm-grid-map-projections/utm-grid-universal-transverse-mercator-projection/9779}{Natural Resources of Canada}.}
}

\newglossaryentry{odc}{
    name=Ontario Data Catalogue,
    description={An online platform providing public access to datasets and information from the Government of Ontario. Explore the catalogue \href{https://data.ontario.ca/}{here}. Ontario Data Catalogue provides an API allowing developers to programmatically search, access, and retrieve datasets from the Ontario Data Catalogue. Learn more it \href{https://data.ontario.ca/about}{here}.}
}

\newglossaryentry{od}{
    name=Open Data,
    description={Data made freely available to the public, with no restrictions on use, modification, or sharing. The Ontario Data Catalogue focuses on publishing open data to foster transparency and innovation.}
}

\newglossaryentry{gc}{
    name=Grid Capacity,
    description={The maximum amount of electricity that can be reliably transmitted and distributed through the power grid to meet demand. It is influenced by factors such as infrastructure, generation sources, and peak usage times.}
}

\newglossaryentry{sc}{
    name=Statistics Canada,
    description={A platform that provides access to a wide range of Canadian statistical data and metadata. It supports research, analysis, and decision-making in various fields. Explore datasets and tools \href{https://www.statcan.gc.ca/}{here}.}
}

\newglossaryentry{grt-open}{
    name=Grand River Transit Open Data,
    description={An online resource offering access to transit-related datasets, including routes, schedules, and real-time data for the Grand River area. Useful for developers and analysts working on transportation projects. Access the portal \href{https://www.grt.ca/en/about-grt/open-data.aspx}{here}.}
}

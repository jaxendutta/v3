\newacronym[
  description={Artificial Intelligence: a branch of computer science concerned with
  building systems that perform tasks typically requiring human intelligence; used in
  domestic social robots for natural language processing, facial recognition, emotion
  detection, and adaptive behavioral responses}
]{ai}{AI}{Artificial Intelligence}

\newacronym[
  description={Human-Robot Interaction: the interdisciplinary field studying how humans
  and robots communicate, collaborate, and relate; central to analyzing how anthropomorphic
  design shapes user expectations, trust, and behavior toward robotic systems}
]{hri}{HRI}{Human-Robot Interaction}

\newacronym[
  description={Value Sensitive Design: a methodological framework developed by Batya
  Friedman and colleagues for proactively incorporating human values with ethical import
  throughout the technology design process, through iterative conceptual, empirical,
  and technical investigations}
]{vsd}{VSD}{Value Sensitive Design}

\newacronym[
  description={Light-emitting diode: a semiconductor light source used in the proposed
  Spatial Privacy Zones feature as a visible physical indicator of \neo{}'s sensor state,
  signaling to users when cameras and microphones have been deactivated in designated
  privacy zones}
]{led}{LED}{Light-Emitting Diode}

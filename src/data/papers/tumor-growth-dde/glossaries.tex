\newglossaryentry{cytotoxic}
{
name=cytotoxic,
description={Toxic to cells.}
}

\newglossaryentry{relapse}
{
name=relapse,
description={Slip back into a previous state of illness.}
}

\newglossaryentry{growth arrest}
{
name=growth arrest,
description={A stopping point in the cell cycle, where it is no longer involved in the processes surrounding duplication and division.}
}

\newglossaryentry{chemotherapy}
{
name=chemotherapy,
description={Treatment that uses drugs to stop the growth of cancer cells, either by killing the cells or by stopping them from dividing.}
}

\newglossaryentry{half-life}
{
name=half-life,
description={The length of time required for the concentration of a particular substance (typically a drug) to decrease to half of its starting dose in the body.}
}

\newglossaryentry{Michaelis-Menten term}
{
name=Michaelis-Menten term,
description={Mainly used to characterize the enzymatic rate at different substrate concentrations, but it is also widely applied to characterize the elimination of chemical (the first-order kinetics) compounds from the body.}
}

\newglossaryentry{Hopf bifurcation}
{
name=Hopf bifurcation,
description={The local birth or death of a periodic solution (self-excited oscillation) from equilibrium as a parameter crosses a critical value.}
}

\newglossaryentry{stochastic}
{
name=stochastic,
description={Involving a random variable.}
}

\newglossaryentry{deterministic}
{
name=deterministic,
description={No randomness is involved in the development of future states.}
}

\newglossaryentry{spatial heterogeneity}
{
name=spatial heterogeneity,
description={The existence of differences in characteristics such as genetic information and cell morphology.}
}

\newglossaryentry{basin of attraction}
{
name=basin of attraction,
description={The region of the phase space, over which iterations are defined, such that any point (any initial condition) in that region will asymptotically be iterated into the attractor.}
}

\newglossaryentry{bioavailability }
{
name=bioavailability,
description={The ability of a drug or other substance to be absorbed and used by the body.}
}

\newglossaryentry{deterministic system}
{
name=deterministic system,
description={A system in which no randomness is involved in the development of future states.}
}
